Group 1: Search Methods

Direct Search (Start ‚Üí End)

1. Array.indexOf()

Description: Finds the first index of an element.

js 
const arr = [2, 5, 2, 7];
console.log(arr.indexOf(2)); // 0

Equivalent for loop:

js 
function indexOf(array, value) {
  for (let i = 0; i < array.length; i++) {
    if (array[i] === value) return i;
  }
  return -1;
}

2. Array.includes()

Description: Checks if an element exists.

js 
console.log(arr.includes(5)); // true

Equivalent for loop:

js 
function includes(array, value) {
  for (let i = 0; i < array.length; i++) {
    if (array[i] === value) return true;
  }
  return false;
}

3. Array.find()

Description: Returns the first element that satisfies a condition.

js 
console.log(arr.find(num => num > 4)); // 5

Equivalent for loop:

js 
function find(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i, array)) return array[i];
  }
  return undefined;
}

4. Array.findIndex()

Description: Returns the first index that satisfies a condition.

js 
console.log(arr.findIndex(num => num > 4)); // 1

Equivalent for loop:

js 
function findIndex(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i, array)) return i;
  }
  return -1;
}

console.log(arr.lastIndexOf(2)); // 2

function lastIndexOf(array, value) {
  for (let i = array.length - 1; i >= 0; i--) {

function findLast(array, callback) {

function findLastIndex(array, callback) {

Condition Verification

8. Array.every()

Description: Checks if every element satisfies a condition.

js 
console.log(arr.every(num => num > 1)); // false

Equivalent for loop:

js 
function every(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (!callback(array[i], i, array)) return false;
  }
  return true;
}

9. Array.some()

Description: Checks if at least one element satisfies a condition.

js 
console.log(arr.some(num => num > 6)); // true

Equivalent for loop:

js 
function some(array, callback) {
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i, array)) return true;
  }
  return false;
}

Comparisons and Use Cases:

  • indexOf(): Returns the first index ‚Üí To find an element's position.
  • lastIndexOf(): Returns the last index ‚Üê To find the last occurrence.
  • includes(): Boolean ‚Üí For quick existence checks.
  • find()/findLast(): Returns the element ‚Üí For finding complex objects.
  • findIndex()/findLastIndex(): Returns the index ‚Üí For finding object positions.
  • every(): Boolean ‚Üí For complete data validation.
  • some(): Boolean ‚Üí To check if at least one element is valid.

Advanced Examples:

js 
// find() for objects:
const users = [{id: 1}, {id: 2}];
console.log(users.find(user => user.id === 2)); // {id: 2}

// every() for validation:
const values = [15, 20, 25];
console.log(values.every(age => age >= 18)); // false

// some() with strings:
const words = ["JS", "Python"];
console.log(words.some(p => p.includes("Py"))); // true

Group 2: Array Manipulation

Adding/Removing at the End

1. Array.push()

Description: Adds elements to the end of the array.

js 
const arr = [1, 2];
arr.push(3, 4); // [1, 2, 3, 4]

Equivalent for loop:

js 
function push(array, ...elements) {
  for (let i = 0; i < elements.length; i++) {
    array[array.length] = elements[i];
  }
  return array.length;
}

2. Array.pop()

Description: Removes the last element.

js 
arr.pop(); // [1, 2, 3]

Equivalent for loop:

js 
function pop(array) {
  if (array.length === 0) return undefined;
  const removed = array[array.length - 1];
  array.length--;
  return removed;
}

Adding/Removing at the Beginning

3. Array.shift()

Description: Removes the first element.

js 
arr.shift(); // [2, 3]

Equivalent for loop:

js 
function shift(array) {
  if (array.length === 0) return undefined;
  const removed = array[0];
  for (let i = 0; i < array.length - 1; i++) {
    array[i] = array[i + 1];
  }
  array.length--;
  return removed;
}

4. Array.unshift()

Description: Adds elements at the beginning.

js 
arr.unshift(0); // [0, 1, 2, 3]

Equivalent for loop:

js 
function unshift(array, ...elements) {
  for (let i = array.length + elements.length - 1; i >= elements.length; i--) {
    array[i] = array[i - elements.length];
  }
  for (let j = 0; j < elements.length; j++) {
    array[j] = elements[j];
  }
  return array.length;
}

Specific Position Modification

5. Array.splice()

Description: Adds or removes elements at any position.

js 
arr.splice(1, 2, "a", "b"); // Removes 2 elements from index 1 and adds "a", "b"

Equivalent for loop:

js 
function splice(array, start, deleteCount, ...items) {
  const removed = [];
  // Store removed elements
  for (let i = start; i < start + deleteCount; i++) {
    removed.push(array[i]);
  }
  // Shift remaining elements
  const newLength = array.length - deleteCount + items.length;
  for (let i = newLength - 1; i >= start + items.length; i--) {
    array[i] = array[i - items.length + deleteCount];
  }
  // Insert new items
  for (let j = 0; j < items.length; j++) {
    array[start + j] = items[j];
  }
  array.length = newLength;
  return removed;
}

Order Modification

6. Array.reverse()

Description: Reverses the order of the array elements.

js 
arr.reverse(); // [3, 2, 1]

Equivalent for loop:

js 
function reverse(array) {
  for (let i = 0; i < Math.floor(array.length / 2); i++) {
    const temp = array[i];
    array[i] = array[array.length - 1 - i];
    array[array.length - 1 - i] = temp;
  }
  return array;
}

7. Array.sort()

Description: Sorts the array elements (by default, as strings).

js 
arr.sort(); // Alphabetical or numerical sorting

Equivalent for loop (basic Bubble Sort):

js 
function sort(array) {
  for (let i = 0; i < array.length; i++) {
    for (let j = 0; j < array.length - 1; j++) {
      if (String(array[j]) > String(array[j + 1])) {
        const temp = array[j];
        array[j] = array[j + 1];
        array[j + 1] = temp;
      }
    }
  }
  return array;
}

Filling and Copying

8. Array.fill()

Description: Fills the array with a specified value.

js 
arr.fill(0, 1, 3); // [1, 0, 0, 4]

Equivalent for loop:

js 
function fill(array, value, start = 0, end = array.length) {
  for (let i = start; i < end; i++) {
    array[i] = value;
  }
  return array;
}

9. ArrayWithin()

Description: Copies elements within the same array.

js 
// Copies elements from indices 2-3 to index 0
within(arr, 0, 2, 4);

Equivalent for loop:

js 
function within(array, target, start, end = array.length) {
  const copied = [];
  for (let i = start; i < end; i++) copied.push(array[i]);
  for (let j = 0; j < copied.length; j++) {
    array[target + j] = copied[j];
  }
  return array;
}

Comparisons and Use Cases:

  • push()/pop(): Adding/removing at the end (Stack ‚Äì LIFO)
  • shift()/unshift(): Adding/removing at the beginning (Queue ‚Äì FIFO)
  • splice(): Modification at specific positions (Dynamic replacement)
  • reverse(): Reverses order (Reverse display)
  • sort(): Sorting (Data organization)
  • fill(): Filling (Array initialization)
  • within(): Internal copying (Buffer manipulation)

Practical Examples:

js 
// Using push() to accumulate data:
const logs = [];
logs.push("Error 404", "Warning"); // Adds logs at the end

// Using splice() for replacement:
const fruits = ["üçé", "üçå", "üçá"];
fruits.splice(1, 1, "ü•≠"); // Removes "üçå" and adds "ü•≠"

// Using within() for optimization:
const buffer = [10, 20, 30, 40];
within(buffer, 0, 2); // [30, 40, 30, 40]

// Using fill() for initialization:
const matrix = new Array(3).fill().map(() => new Array(3).fill(0));
// Creates a 3x3 matrix filled with 0

Note: The for loop implementations are simplified to demonstrate the logic. Native methods are more optimized and handle edge cases.

Group 3: Array Creation and Transformation

Creating New Arrays (Without Modifying the Original)

1. Array.concat()

Description: Combines arrays into a new array.

js 
const arr1 = [1, 2];
const arr2 = [3, 4];
const merged = arr1.concat(arr2); // [1, 2, 3, 4]

Equivalent for loop:

js 
function concat(array, ...arrays) {
  const novo = [...array];
  for (const arr of arrays) {
    for (let i = 0; i < arr.length; i++) {
      novo.push(arr[i]);
    }
  }
  return novo;
}

2. Array.slice()

Description: Extracts a portion of the array into a new array.

js 
const subArr = arr.slice(1, 3); // [2, 3]

Equivalent for loop:

js 
function slice(array, start = 0, end = array.length) {
  const novo = [];
  for (let i = start; i < end && i < array.length; i++) {
    novo.push(array[i]);
  }
  return novo;
}

3. Array.with()

Description: Creates a new array with one element replaced at a given index.

js 
const novoArr = arr.with(1, 99); // [1, 99, 3]

Equivalent for loop:

js 
function withMethod(array, index, value) {
  const novo = [...array];
  novo[index] = value;
  return novo;
}

Nested Array Manipulation

4. Array.flat()

Description: Flattens nested arrays.

js 
const arrAninhado = [1, [2, [3]]];
const flatArr = arrAninhado.flat(2); // [1, 2, 3]

Equivalent for loop (for depth 1):

js 
function flat(array) {
  const novo = [];
  for (let i = 0; i < array.length; i++) {
    if (Array.isArray(array[i])) {
      for (const item of array[i]) novo.push(item);
    } else {
      novo.push(array[i]);
    }
  }
  return novo;
}

5. Array.flatMap()

Description: Maps and flattens the result (equivalent to map() followed by flat(1)).

js 
const duplicated = arr.flatMap(num => [num, num * 2]); // [1, 2, 2, 4]

Equivalent for loop:

js 
function flatMap(array, callback) {
  const novo = [];
  for (let i = 0; i < array.length; i++) {
    const result = callback(array[i], i, array);
    if (Array.isArray(result)) {
      for (const item of result) novo.push(item);
    } else {
      novo.push(result);
    }
  }
  return novo;
}

Immutable Copy Methods (ES2023+)

6. Array.toReversed()

Description: Creates a reversed copy of the array (without modifying the original).

js 
const reversed = arr.toReversed(); // [3, 2, 1]

Equivalent for loop:

js 
function toReversed(array) {
  const novo = [];
  for (let i = array.length - 1; i >= 0; i--) {
    novo.push(array[i]);
  }
  return novo;
}

7. Array.toSorted()

Description: Creates a sorted copy of the array.

js 
const sorted = arr.toSorted((a, b) => b - a); // [3, 2, 1]

Equivalent for loop (basic simulation):

js 
function toSorted(array, compareFn) {
  const novo = [...array];
  return novo.sort(compareFn);
}

8. Array.toSpliced()

Description: Creates a copy of the array with elements added or removed.

js 
const spliced = arr.toSpliced(1, 1, 99); // [1, 99, 3]

Equivalent for loop:

js 
function toSpliced(array, start, deleteCount, ...items) {
  const novo = [...array];
  novo.splice(start, deleteCount, ...items);
  return novo;
}

Comparisons and Use Cases:

  • concat(): Combines data from multiple sources.
  • slice(): Safely extracts subarrays (useful for pagination).
  • flat(): Flattens nested arrays (e.g., JSON).
  • flatMap(): Processes and flattens simultaneously.
  • toReversed(): Creates a reversed copy without modifying the original.
  • toSorted(): Provides immutable sorting for display purposes.
  • with(): Immutable element update.
  • toSpliced(): Immutable update with addition or removal of elements.

Practical Examples:

js 
// Using flatMap() to expand data:
const phrases = ["Hello world", "JS is cool"];
const words = phrases.flatMap(phrase => phrase.split(" "));
// Result: ["Hello", "world", "JS", "is", "cool"]

// Using toSorted() for safe sorting:
const numbers = [3, 1, 2];
const sorted = numbers.toSorted((a, b) => a - b); // [1, 2, 3]
// The original array remains [3, 1, 2]

// Using with() for immutable updates:
const cart = ["üçé", "üçå"];
const newCart = cart.with(1, "ü•ù"); // ["üçé", "ü•ù"]

// Using toSpliced() for element replacement:
const months = ["Jan", "Feb", "Apr"]; // "Mar" is missing
const corrected = months.toSpliced(2, 0, "Mar"); // Inserts "Mar" at index 2

Note: The methods toReversed(), toSorted(), toSpliced(), and with() are new in ES2023 and do not modify the original array.

Group 4: Iteration and Accumulation

Iteration and Transformation

1. Array.forEach()

Description: Executes a function for each element.

js 
const arr = [1, 2, 3];
arr.forEach(num => console.log(num * 2)); // Prints: 2, 4, 6

Equivalent for loop:

js 
function forEach(array, callback) {
  for (let i = 0; i < array.length; i++) {
    callback(array[i], i, array);
  }
}

2. Array.map()

Description: Creates a new array by transforming each element.

js 
const doubled = arr.map(num => num * 2); // [2, 4, 6]

Equivalent for loop:

js 
function map(array, callback) {
  const novo = [];
  for (let i = 0; i < array.length; i++) {
    novo.push(callback(array[i], i, array));
  }
  return novo;
}

3. Array.filter()

Description: Filters the array based on a condition.

js 
const evens = arr.filter(num => num % 2 === 0); // [2]

Equivalent for loop:

js 
function filter(array, callback) {
  const novo = [];
  for (let i = 0; i < array.length; i++) {
    if (callback(array[i], i, array)) novo.push(array[i]);
  }
  return novo;
}

Reduction and Accumulation

4. Array.reduce()

Description: Reduces the array to a single value (from left to right).

js 
const sum = arr.reduce((acc, num) => acc + num, 0); // 6

Equivalent for loop:

js 
function reduce(array, callback, initial) {
  let acc = initial !== undefined ? initial : array[0];
  let start = initial !== undefined ? 0 : 1;
  for (let i = start; i < array.length; i++) {
    acc = callback(acc, array[i], i, array);
  }
  return acc;
}

5. Array.reduceRight()

Description: Reduces the array from right to left.

js 
const reversedStr = ["m", "u", "n"].reduceRight((acc, letter) => acc + letter, ""); // "num"

Equivalent for loop:

js 
function reduceRight(array, callback, initial) {
  let acc = initial !== undefined ? initial : array[array.length - 1];
  let start = initial !== undefined ? array.length - 1 : array.length - 2;
  for (let i = start; i >= 0; i--) {
    acc = callback(acc, array[i], i, array);
  }
  return acc;
}

Group 5: Array Iterators

Iterators

6. Array.entries()

Description: Returns an iterator of [index, value] pairs.

js 
const iterator = arr.entries();
console.log(iterator.next().value); // [0, 1]

For loop simulation of the iterator:

js 
function entries(array) {
  let index = 0;
  return {
    next: () => {
      return index < array.length 
        ? { value: [index, array[index++]], done: false } 
        : { done: true };
    }
  };
}

7. Array.keys()

Description: Returns an iterator of the array's indices.

js 
for (const key of arr.keys()) console.log(key); // 0, 1, 2

For loop simulation of the iterator:

js 
function keys(array) {
  let index = 0;
  return {
    next: () => {
      return index < array.length 
        ? { value: index++, done: false } 
        : { done: true };
    }
  };
}

8. Array.values()

Description: Returns an iterator of the array's values.

js 
for (const value of arr.values()) console.log(value); // 1, 2, 3

For loop simulation of the iterator:

js 
function values(array) {
  let index = 0;
  return {
    next: () => {
      return index < array.length 
        ? { value: array[index++], done: false } 
        : { done: true };
    }
  };
}

9. Array[Symbol.iterator]()

Description: The default iterator of the array (equivalent to values()).

js 
const iterator = arr[Symbol.iterator]();
console.log(iterator.next().value); // 1

Implementation: Same as the values() function above.

Comparisons and Use Cases:

  • forEach(): Executes actions for side effects (e.g., logging).
  • map(): Creates a new transformed array.
  • filter(): Creates a new filtered array.
  • reduce()/reduceRight(): Aggregate values (sum, concatenation).
  • entries(), keys(), values(), [Symbol.iterator](): Facilitate iteration with access to indices and values.

Practical Examples:

js 
// Using map() to format data:
const prices = [10, 20, 30];
const formatted = prices.map(p => `R$${p},00`); // ["R$10,00", "R$20,00", "R$30,00"]

// Using reduce() to group objects:
const people = [{name: "Ana"}, {name: "John"}];
const names = people.reduce((acc, p) => acc + p.name + " ", ""); // "Ana John "

// Using entries() for detailed loop:
for (const [i, value] of arr.entries()) {
  console.log(`Index ${i}: ${value}`);
}

// Using reduceRight() to reverse a string:
const str = "abc";
const reversedStr = str.split("").reduceRight((acc, char) => acc + char, ""); // "cba"

Group 6: String Manipulation

String Methods

1. String.length

Description: Returns the number of characters in the string.

js 
const str = "Hello";
console.log(str.length); // 5

Equivalent for loop:

js 
function length(str) {
  let count = 0;
  for (const _ of str) count++;
  return count;
}

2. String.charAt()

Description: Returns the character at the specified index.

js 
console.log(str.charAt(1)); // "e"

Equivalent for loop:

js 
function charAt(str, index) {
  for (let i = 0; i < str.length; i++) {
    if (i === index) return str[i];
  }
  return "";
}

3. String.charCodeAt()

Description: Returns the Unicode code of the character at the specified index.

js 
console.log(str.charCodeAt(0)); // 72 (for "H")

Equivalent for loop:

js 
function charCodeAt(str, index) {
  for (let i = 0; i < str.length; i++) {
    if (i === index) return str.charCodeAt(i);
  }
  return NaN;
}

4. String.at()

Description: Returns the character at the specified index, supporting negative indices.

js 
console.log(str.at(-1)); // "o"

Equivalent for loop:

js 
function at(str, index) {
  const adjustedIndex = index < 0 ? str.length + index : index;
  for (let i = 0; i < str.length; i++) {
    if (i === adjustedIndex) return str[i];
  }
  return undefined;
}

5. String.slice()

Description: Extracts a portion of the string.

js 
console.log(str.slice(1, 3)); // "el"

Equivalent for loop:

js 
function slice(str, start, end = str.length) {
  let substring = "";
  for (let i = start; i < end && i < str.length; i++) {
    substring += str[i];
  }
  return substring;
}

6. String.substring()

Description: Similar to slice(), but swaps indices if needed.

js 
console.log(str.substring(3, 1)); // "el"

Equivalent for loop:

js 
function substring(str, start, end) {
  const [s, e] = start > end ? [end, start] : [start, end];
  return slice(str, s, e);
}

7. String.substr() (Deprecated)

Description: Extracts a substring from a given index with a specified length.

js 
console.log(str.substr(1, 2)); // "el"

Equivalent for loop:

js 
function substr(str, start, length) {
  let substring = "";
  for (let i = start; i < start + length && i < str.length; i++) {
    substring += str[i];
  }
  return substring;
}

8. String.toUpperCase()

Description: Converts the string to uppercase.

js 
console.log(str.toUpperCase()); // "HELLO"

Equivalent for loop:

js 
function toUpperCase(str) {
  let upper = "";
  for (const char of str) {
    const code = char.charCodeAt(0);
    upper += code >= 97 && code <= 122 ? String.fromCharCode(code - 32) : char;
  }
  return upper;
}

9. String.toLowerCase()

Description: Converts the string to lowercase.

js 
console.log("HELLO".toLowerCase()); // "hello"

Equivalent for loop:

js 
function toLowerCase(str) {
  let lower = "";
  for (const char of str) {
    const code = char.charCodeAt(0);
    lower += code >= 65 && code <= 90 ? String.fromCharCode(code + 32) : char;
  }
  return lower;
}

10. String.concat()

11. String.trim()

Description: Removes whitespace from the beginning and end of the string.

js

Equivalent for loop:

js 
function trim(str) {
  let start = 0, end = str.length - 1;
  while (str[start] === " ") start++;
  while (str[end] === " ") end--;
  let trimmed = "";
  for (let i = start; i <= end; i++) trimmed += str[i];
  return trimmed;
}

13. String.padStart()

Description: Pads the string at the beginning until it reaches a specified length.

js

Equivalent for loop:

js 
function padStart(str, targetLength, padString = " ") {
  let padded = str;
  while (padded.length < targetLength) {
    padded = padString + padded;
  }
  return padded.slice(-targetLength);
}

14. String.padEnd()

Description: Pads the string at the end until it reaches a specified length.

Equivalent for loop:

js 
function padEnd(str, targetLength, padString = " ") {
  let padded = str;

16. String.replace()

Description: Replaces the first occurrence of a pattern with another string.

js 
console.log("Hello World".replace("World", "JS")); // "Hello JS"

Equivalent for loop:

js 
function replace(str, search, replacement) {
  let result = "";
  let replaced = false;
  for (let i = 0; i < str.length; ) {
    if (!replaced && str.slice(i, i + search.length) === search) {
      result += replacement;
      i += search.length;
      replaced = true;
    } else {
      result += str[i];
      i++;
    }
  }
  return result;
}

    }

18. String.split()

Description: Splits the string into an array based on a separator.

js 
console.log("a,b,c".split(",")); // ["a", "b", "c"]

Equivalent for loop:

js 
function split(str, separator) {
  const arr = [];
  let current = "";
  for (const char of str) {
    if (char === separator) {
      arr.push(current);
      current = "";
    } else {
      current += char;
    }
  }
  arr.push(current);
  return arr;
}

Comparisons and Use Cases:

  • slice(): Flexible extraction, supports negative indices.
  • substring(): Extraction without negative indices.
  • padStart(): Useful for formatting (e.g., leading zeros).
  • replaceAll(): Text cleaning (replace multiple occurrences).
  • split(): Processing CSV/TSV data.
  • trim(): Cleaning user input.
